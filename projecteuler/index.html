Problème 1 

x = 0
for i in range(1000):
    if i%3 == 0 or i%5 == 0:
        x += i

Problème 2 

i = 0
somme = [2]
chx = 1
chn = 2
while i < 4000000:
    i = chx + chn
    chx = chn
    chn = i
    if i < 4000000 and i%2 == 0:
        somme.append(i)
print(somme)
print(sum(somme))
